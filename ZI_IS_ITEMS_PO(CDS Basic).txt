define view entity ZI_IS_ITEMS_PO
  as select from I_SalesDocumentItem

  association [0..1] to I_ProductDescription_2 as _ProductDesc on  $projection.Material  = _ProductDesc.Product
                                                               and _ProductDesc.Language = 'P'

{
  key SalesDocument                   as Pedido,
  
      @UI.lineItem: [ { position: 10, label: 'Item Pedido' } ]
  key SalesDocumentItem               as ItemPedido,

      @UI.lineItem: [ { position: 20, label: 'Material' } ]
      Material                        as Material,

      @UI.lineItem: [ { position: 30, label: 'Descrição' } ]
      _ProductDesc.ProductDescription as DescricaoMaterial,

      @UI.lineItem: [ { position: 40, label: 'Centro' } ]
      Plant                           as Centro,

      @Semantics.quantity.unitOfMeasure: 'UnidadeMedida'
      @UI.lineItem: [ { position: 50, label: 'Quantidade' } ]
      OrderQuantity                   as Quantidade,

      @UI.lineItem: [ { position: 60, label: 'Unidade Medida' } ]
      OrderQuantityUnit               as UnidadeMedida,

      @Semantics.amount.currencyCode: 'Moeda'
      @UI.lineItem: [ { position: 70, label: 'Valor Liquido' } ]
      NetAmount                       as ValorLiquido,

      @UI.lineItem: [ { position: 80, label: 'Moeda' } ]
      TransactionCurrency             as Moeda,

      SalesDocumentItemText           as TextoItem,

      @UI.lineItem: [ { position: 90, label: 'Status Entrega' } ]
      DeliveryStatus                  as StatusEntrega,

      @UI.lineItem: [ { position: 100, label: 'Status Processo' } ]
      SDProcessStatus                 as StatusProcesso,

      _ProductDesc

}
