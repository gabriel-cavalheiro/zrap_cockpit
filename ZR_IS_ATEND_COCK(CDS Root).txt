@AccessControl.authorizationCheck: #MANDATORY
@EndUserText.label: 'Reuse - Atendimento Cockpit'
@ObjectModel.sapObjectNodeType.name: 'ZTBIS_ATEND_COCK'
@VDM.viewType: #TRANSACTIONAL

define root view entity ZR_IS_ATEND_COCK
  as select from ztbis_atend_cock

  association [0..1] to I_Customer    as _Customer    on $projection.Customer = _Customer.Customer
  association [0..1] to I_CompanyCode as _CompanyCode on $projection.Companycode = _CompanyCode.CompanyCode
  association [0..*] to ZI_IS_ITEMS_PO as _ItemsPO    on $projection.Pedido = _ItemsPO.Pedido

{
  key atendimento_id                                           as AtendimentoID,

      companycode                                              as Companycode,
      customer                                                 as Customer,

      cast( _Customer.CustomerName as abap.char(60) )          as CustomerName,
      situacao                                                 as Situacao,
      data_atendimento                                         as DataAtendimento,
      operador                                                 as Operador,
      nome_operador                                            as NomeOperador,
      operacao                                                 as Operacao,
      descr_operacao                                           as DescrOperacao,
      tipo_ocorrencia                                          as TipoOcorrencia,
      descr_ocorrencia                                         as DescrOcorrencia,
      pedido                                                   as Pedido,
      nfe                                                      as Nfe,
      observacao                                               as Observacao,

      // Criticidade por situação
      case situacao
        when 'AB' then 2   // Aberto = Amarelo
        when 'FI' then 3   // Finalizado = Verde
        when 'CA' then 1   // Cancelado = Vermelho
        else 0
      end                                                      as SituacaoCriticality,
      
      case situacao
        when 'AB' then 'Em Aberto'  
        when 'FI' then 'Finalizado'
        when 'CA' then 'Cancelado'
        else ''
      end                                                      as LabelSituacao,

      concat_with_space(
      'Nº do Atendimento:', ltrim( atendimento_id, '0' ), 1  ) as LabelTitle,

      @Semantics.user.createdBy: true
      local_created_by                                         as LocalCreatedBy,

      @Semantics.systemDateTime.createdAt: true
      local_created_at                                         as LocalCreatedAt,

      @Semantics.user.localInstanceLastChangedBy: true
      local_last_changed_by                                    as LocalLastChangedBy,

      @Semantics.systemDateTime.localInstanceLastChangedAt: true
      local_last_changed_at                                    as LocalLastChangedAt,

      @Semantics.systemDateTime.lastChangedAt: true
      last_changed_at                                          as LastChangedAt,

      // Associations
      _Customer,
      _CompanyCode,
      _ItemsPO

}
